CREATE DATABASE CODEFLIP;
USE CODEFLIP;

CREATE TABLE USER(
USERNAME VARCHAR(40),
EMAIL VARCHAR(50) NOT NULL PRIMARY KEY,
PASSWORD VARCHAR(100),
BIO TEXT
);

CREATE TABLE CARD(
TITLE VARCHAR(50),
CODE TEXT NOT NULL,
CREATED_BY VARCHAR(50) NOT NULL,
ID_ INT AUTO_INCREMENT PRIMARY KEY,
LANGUAGE VARCHAR(30)
);

CREATE TABLE user (
    username CHAR(40),
    email CHAR(50) NOT NULL PRIMARY KEY,
    password CHAR(100),
    bio TEXT
);
